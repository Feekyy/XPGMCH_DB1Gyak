CREATE TABLE Tevékenység(TKód int primary key, bizonyított int, feltételezett int not null, első_időpont int, utolsó_időpont year);
CREATE TABLE Személyek(SzKód int primary key, név char(50), szül_idő int, hal_idő int, típus char(50), csoport_név char(50), TKód int REFERENCES Tevékenység.TKód);
create table Becenév(BKód int primary key, SzKód int references személyek.SzKód, becenév char(50));
create table Büntetés(BKód int primary key, elfogás_ideje int, büntetés_megkezdése int, büntetés_befejezése int, ország char(50), típus char(50));
create table Helyszín(HKód int primary key, ország char(50), megye int, betört char(10));
create table H_Sz(HKód int references Helyszín.HKód, SzKód int references Személyek.SzKód, primary key(HKód, SzKód));
create table Elkövetés_módja(EKód int primary key, célpont char(50), erőszakolt char(5), kínzott char(5), napszak char(10));
create table Eszköz(EKód int primary key, EMKód int references elkövetés_módja.EKód, eszköz char(20));
create table Érdekességek(ÉKód int primary key, elkövetés_indítéka char(50), kivégezték char(4), vallott char(5));
create table Megye(MKód int references helyszín.megye, megye char(50));
