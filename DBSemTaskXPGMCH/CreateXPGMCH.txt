CREATE TABLE Tevékenység(TKód int primary key, bizonyított int, feltételezett int not null, első idópont date, utolsó időpont);
CREATE TABLE Személyek(SzKód int primary key, név chr(50), szül. idő date, hal. idő date, típus char(10), csoport név char(50), TKód in REFERENCES Tevékenység.TKód);
CREATE TABLE Becenév(Bkód int references Személyek.SzKód, becenév char(50));
create table Intézmény(IKód int primary key, ország char(50), város char(50), típus char(10));
create table Büntetés(BKód int primary key, elfogás ideje date, büntetés megkezdése date, büntetés befejezése date, IKód int references Intézmény.IKód);
create table Helyszín(HKód int primary key, ország char(50), város char(50), városrész char(50));
create table H_Sz(HKód int references Helyszín.HKód, SzKód int references Személyek.SzKód);
create table Elkövetés módja(EKód int primary key, nem char(3), napszak char(5));
create table Eszköz(EKód int references EKód, eszköz char(20));
create table Érdekességek(ÉKód int primary key, erőszakolt char(4), elkövetés indítéka char(50), kivégezték char(4), kínzott char(4), betört char(4));
